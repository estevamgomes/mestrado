/*
 * Input Analog > Output Servo
 * 
 * Description: Reads an analog sensor that controls a servo motor 
 * 
 * Recommended inputs: Potentiometer
 * Recommended outputs: Servo Motor
 *

 */

#include <Servo.h>  // include the servo library
Servo myservo;      // create servo object to control a servo

const int analogInPin  = A0;  // the number of the analog input pin
const int analogOutPin = 9;   // the number of the pwm output pin

int sensorValue = 0;          // store the value read from analog input
int outputValue = 0;          // store the value to output to the servo (analog output)

void setup() {
  myservo.attach(analogOutPin);  // attaches the servo on pin 9 to the servo object
}

void loop() {
  sensorValue = analogRead(analogInPin);            // read the analog in value
  outputValue = map(sensorValue, 0, 1023, 0, 180);  // map it from the input range (0-1023) to the output range (0-180)
  myservo.write(outputValue);                       // sets the servo position according to the scaled input
  delay(15);                                        // waits for the servo to complete the movement
}
